# BUILD MACHINES
docker-compose up
docker-compose -f stack.yml up

# TO EXECUTE COMMANDS INSIDE OF CONTAINER
docker exec -it env_db_1 /bin/bash
docker exec -it env_adminer_1 /bin/bash

# TO INSPECT DATA IN  DOCKER IMAGES
docker images
docker ps -a
docker inspect e47c589b5151
docker inspect env_db_1 | grep IPAddress

# TO DELETE ALL DOCKER IMAGES
Docker images -q | xargs sudo docker rmi -f

# TO DELETE ALL DOCKER CONTAINERS
docker ps -aq | xargs sudo docker rm -f

# GET POSTGRES SERVER IP ADDRESS
sudo docker inspect env_db_1 | grep IPAddress
