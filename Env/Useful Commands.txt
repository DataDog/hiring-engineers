Standalone
========

docker-compose up
docker-compose -f stack.yml up

# TO EXECUTE COMMANDS INSIDE OF CONTAINER
docker exec -it <container name> /bin/bash

# TO INSPECT DATA IN  DOCKER IMAGES
docker images
docker ps -a
docker inspect e47c589b5151
docker inspect mysql_1 | grep IPAddress

# TO DELETE ALL DOCKER IMAGES
Docker images -q | xargs sudo docker rmi -f

# TO DELETE ALL DOCKER CONTAINERS
docker ps -aq | xargs sudo docker rm -f

# THE FOLLOWING COMMANDS USE PATHS RELATIVE TO THE FOLDER CONTAINING THIS FILE

# CREATE MYSQL DATA VOLUME
sudo docker create --name mysqldata_1 -v ${PWD}/volumes/mysqldata busybox

# ACCESS POSTGRES'S CONTAINER
sudo docker exec -it 0setuptheenvironment_db_1 bash

# GET POSTGRES SERVER IP ADDRESS
sudo docker inspect 0setuptheenvironment_db_1 | grep IPAddress
